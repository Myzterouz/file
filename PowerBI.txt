    IF(
        CNT > 0,

    IF(
        MAX(Yield_tbl[Process]) = "TARGET",
        TARGET,
        IF(
            MAX(Yield_tbl[Process]) = "RESULT",

            PRODUCT(Yield_tbl[Passrate])
